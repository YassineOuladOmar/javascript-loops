The "For" Loop

Syntax

for (voorbeeld 1; voorbeeld 2; voorbeeld 3) {
// code block to be executed
}

Example

for (i = 0; i < 28; i++) {
text +="My age is" + i + " ";
}

The "While" loop

Syntax

while (condition) {
// Wordt uitgevoerd
}

Example

var text = "";
var i = 0;

while (i < 23) {
text +="My age is" + i; i++;
}

The "Do/While" Loop

Syntax

do {
// code block to be executed
} while (condition);

Example

do {
text += "My age is " + i; i++;
}
while (i < 23);

The For/In Loop

Example

var member = {firstname:"Oussama", lastname:"Asbaa", age:22};
var text = "";
var a; for (a in member) {
text += member[a] + " ";
}

The For/Of Loop

Example

var chicks = ['Storm', 'Patrick', 'Nathan'];
var a;
for (a of chicks) {
document.write(a + " ");
}
